\section{Potencia de una relación $R^n$}

\begin {definition}
Given $ R \subseteq A \times A $,

\begin{center}
$ R^n = \begin{cases}
R & n=1 \\
\left\{(a,b) \suchthat \exists x \in A, (a,x)\in R^{n-1}\wedge (x,b)\in R\right\} & n > 1

\end{cases}$
\end{center}

\end {definition}

\begin{example}
Given the binary relation $R = \{(1, 1), (1, 2), (2, 3),(3,4)\}$


\qquad $R^3 = \{(1, 1), (1, 2), (1, 3),(1,4)\}$


\end{example}
